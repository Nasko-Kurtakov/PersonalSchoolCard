public static string GetStudentNames(int studentID)
        {
            using (var context = new PersonalSchoolCardEntities())
            {
                var StudentName = context.Students
                    .Where(student => student.StudentID == studentID)
                    .Select(student => new
                    {
                        FirstName = student.FirstName,
                        SecondName = student.SecondName,
                        LastName = student.LastName
                    })
                    .FirstOrDefault()
                    ;

                string FullNames = StudentName.FirstName + " " + StudentName.SecondName
                    + " " + StudentName.LastName;

                return FullNames;

            }
        }

        public static string GetStudentAddress(int studentID)
        {
            using (var context = new PersonalSchoolCardEntities())
            {
                var StudentAddress = context.Students
                    .Where(student => student.StudentID == studentID)
                    .Join(context.Settlements,
                    s => s.SettlementID, st => st.SettlementID,
                    (s, st) => new
                    {
                        Address = s.Address
                    })
                    .FirstOrDefault()
                    ;
                
                return StudentAddress.Address;
            }
        }

        public static string GetStudentSettlement(int studentID)
        {
            using (var context = new PersonalSchoolCardEntities())
            {
                var StudentSettlement = context.Students
                     .Where(student => student.StudentID == studentID)
                    .Join(context.Settlements,
                    o => o.SettlementID, od => od.SettlementID,
                    (o, od) => new
                    {
                        Settlement = od.SettlemnetName
                    })
                    .FirstOrDefault()
                    ;

                string settlement = StudentSettlement.Settlement;

                return settlement;

            }
        }
